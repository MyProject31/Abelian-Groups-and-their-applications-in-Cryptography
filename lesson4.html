RSA Python Implementation

This is a Python implementation of RSA encryption and decryption:

import math

# check ifa given number is prime
def is_prime(x):
    if x < 2:
        return False
    for i in range(2, int(math.sqrt(x)) + 1):
        if x % i == 0:
            return False
    return True

# calculate gcd
def gcd(a, b):
    if b == 0:
        return a
    return gcd(b, a % b)

# modular inverse
def mod_inverse(e, phi):
    d = 1
    while (d * e) % phi != 1:
        d += 1
    return d

p = int(input("Enter a prime number p: "))
q = int(input("Enter a prime number q: "))

if not is_prime(p) or not is_prime(q):
    print("Error: p and q must be prime numbers!")
    exit()

n = p * q
phi = (p - 1) * (q - 1)

e = 3
while gcd(e, phi) != 1:
    e += 2

d = mod_inverse(e, phi)

print("Public key:", (e, n))
print("Private key:", (d, n))

message = input("Enter message: ")
encrypted = []

for ch in message:
    encrypted.append((ord(ch) ** e) % n)

print("Encrypted message:", encrypted)

decrypted = ""

for num in encrypted:
    decrypted += chr((num ** d) % n)

print("Decrypted message:", decrypted)
